01) Criar as tabelas e os relacionamentos do caso Comprobem com base no Anexo 1 – Detalhamento do Modelo Lógico

CREATE TABLE Centro_Custo (
    CodCC VARCHAR(10) PRIMARY KEY,
    NomeCC VARCHAR(50) NOT NULL
);

CREATE TABLE Secao (
    CodSec VARCHAR(10) PRIMARY KEY,
    NomeSec VARCHAR(50) NOT NULL,
    CodCC VARCHAR(10),
    FOREIGN KEY (CodCC) REFERENCES Centro_Custo(CodCC)
);

CREATE TABLE Almoxarife (
    nAlmox INT PRIMARY KEY,
    Almoxarife VARCHAR(50) NOT NULL
);

CREATE TABLE Material (
    CdMat INT PRIMARY KEY,
    DescrMatPrima VARCHAR(100) NOT NULL
);

CREATE TABLE Requisicao (
    NumReq INT PRIMARY KEY,
    DtReq DATE NOT NULL,
    CodSec VARCHAR(10),
    nAlmox INT,
    FOREIGN KEY (CodSec) REFERENCES Secao(CodSec),
    FOREIGN KEY (nAlmox) REFERENCES Almoxarife(nAlmox)
);

CREATE TABLE Req_Mat (
    NumReq INT,
    CdMat INT,
    QtdReq INT NOT NULL,
    VlrUnit DECIMAL(10, 2) NOT NULL,
    PRIMARY KEY (NumReq, CdMat),
    FOREIGN KEY (NumReq) REFERENCES Requisicao(NumReq),
    FOREIGN KEY (CdMat) REFERENCES Material(CdMat)
);
