#Criaçăo da base Biblioteca

create database biblioteca;
use biblioteca;


create table categoria(
 codcategoria integer primary key auto_increment,
 descricao varchar(30),
 tempo integer,
 valMulta float default 0
);


create table associado(
 codAssociado integer primary key auto_increment,
 nome varchar(60) not null,
 cpf varchar(15) not null unique key,
 ra varchar(15),
 categoria integer,
 endereco varchar(80),
 foreign key (categoria) references categoria (codcategoria)
);

create table livro(
 codLivro integer primary key auto_increment,
 titulo varchar(60) not null,
 isbn varchar(15) not null,
 exemplar integer not null,
 assunto varchar(30),
 edicao varchar(3),
 ano varchar(4),
 autor varchar(80), 
 unique key (isbn, exemplar)
);

create table emprestimo(
 codEmprestimo integer primary key auto_increment,
 dat_dev date,
 dat_emp date,
 codAssociado integer not null,
 foreign key (codAssociado) references associado(codAssociado)
);

create table itens (
 codItem integer primary key auto_increment,
 codEMprestimo integer not null,
 codLivro integer not null,
 unique key(codEmprestimo, codLivro),
 foreign key (codEmprestimo) references emprestimo (codEmprestimo),
 foreign key (codLivro) references livro (codLivro)
); 


create table telefones (
 associado integer not null,
 telefone varchar(10) not null,
 foreign key (associado) references associado (codassociado),
 primary key (associado, telefone)
);
