create table sequencial(
id integer primary key auto_increment,
nome varchar(20),
sequencial integer);


#2
delimiter //
create trigger tsequencial
before update on sequencial
for each row
  begin
    if new.sequencial < old.sequencial then
       set new.sequencial = old.sequencial;
    end if;   
  end;
